\section{Contexto e problema}
% -------------------------------------------------%

\begin{frame}{Contexto e problema}
    \begin{itemize}
        \item Anotações em \textit{Markdown} e saídas geradas por agentes de IA crescem rapidamente em volume.
        \item Construir um \emph{segundo cérebro} digital é uma prática consolidada em \emph{Personal Knowledge Management}.
        \item Soberania do dado: o acervo permanece sob controle direto do usuário, em servidor próprio (\textit{self-hosted}).
        \item Interface aberta a múltiplos clientes, humanos e agentes, sem amarrar o conteúdo a um aplicativo único.
        \item Desempenho e escalabilidade internos para acompanhar o crescimento da base.
    \end{itemize}
\end{frame}

% -------------------------------------------------%
\section{Objetivos}
% -------------------------------------------------%

\begin{frame}{Objetivos}
    \textbf{Objetivo geral}
    \begin{itemize}
        \item Desenvolver um servidor \textit{self-hosted} que centralize arquivos em \textit{Markdown} e exponha interface aberta de acesso a usuários e a agentes de IA.
    \end{itemize}

    \vspace{0.5em}
    \textbf{Objetivos específicos}
    \begin{itemize}
        \item Implementar operações de criação, leitura, edição, exclusão e renomeação de notas no servidor.
        \item Oferecer plugin do Obsidian como cliente humano de referência.
        \item Expor uma API REST como única interface de acesso ao servidor.
    \end{itemize}
\end{frame}

\section{Fluxo de uma operação}
% -------------------------------------------------%

\begin{frame}{Fluxo de uma operação}
    \begin{columns}[T]
        \begin{column}{0.42\textwidth}
            \small
            Criação de uma nota:
            \begin{enumerate}\small
                \item Usuário aciona ``criar nota'' no Obsidian.
                \item Plugin envia \texttt{POST /notes} ao servidor.
                \item API valida o \emph{payload} e delega ao repositório.
                \item Repositório grava conteúdo e metadados em uma única transação SQLite.
                \item Servidor responde \texttt{201 Created}; plugin atualiza a interface.
            \end{enumerate}
        \end{column}

\section{Decisões técnicas}
% -------------------------------------------------%

\begin{frame}{Decisões técnicas}
    \begin{itemize}\small
        \item \textbf{Servidor em Go}: alto desempenho e baixo consumo de recursos, mantendo o sistema leve para \textit{self-hosting} em hardware modesto.
        \item \textbf{SQLite como armazenamento}: zero configuração; conteúdo, anexos e metadados gravados em uma única transação; \textit{backup} reduzido a copiar um arquivo.
        \item \textbf{API REST}: contrato HTTP documentado que permite a qualquer cliente, humano ou agente de IA, integrar-se ao sistema.
        \item \textbf{Plugin do Obsidian em TypeScript}: cliente humano de referência; o servidor permanece aberto a outros clientes.
        \item \textbf{Distribuição via Docker}: imagem reprodutível que sobe servidor e banco em conjunto, reduzindo o atrito de instalação.
    \end{itemize}
\end{frame}
